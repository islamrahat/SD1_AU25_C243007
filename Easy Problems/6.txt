/*
    Print numbers from 1 to N, Take N from user input.
*/

import java.util.*;

public class Main {
    public static void main(String[] args) 
    {
        Scanner input = new Scanner(System.in);
        
        int number = input.nextInt();
        for(int i=1; i<=number; i++)
        {
            System.out.println(i);
        }
    }
}
