/*
    Store and print 5 numbers in an array
*/

import java.util.*;

public class Main {
    public static void main(String[] args) 
    {
        Scanner input = new Scanner(System.in);
        
        int[] arr = new int[5];
        for(int i=0; i<5; i++)
        {
            arr[i] = input.nextInt();
        }
        for(int i=0; i<5; i++)
        {
            System.out.printf("%d ", arr[i]);
        }
        System.out.printf("\n");
    }
}
