/*
    Sum of all numbers from 1 to N
*/

import java.util.*;

public class Main {
    public static void main(String[] args) 
    {
        Scanner input = new Scanner(System.in);
        
        int number = input.nextInt();
        long ans = 0;
        for(int i=1; i<=number; i++)
        {
            ans+=i;
        }
        System.out.println("The sum is: "+ans);
    }
}
